select count(distinct(t1.transaction_id)) as payment_count
from transactions as t1
join
transactions as t2
on t1.merchant_id=t2.merchant_id and t1.credit_card_id=t2.credit_card_id and t1.amount=t2.amount
where t1.transaction_timestamp between t2.transaction_timestamp and (t2.transaction_timestamp + interval '10 minutes') and t1.transaction_id !=t2.transaction_id;

select extract(month from u2.event_date) as month,count(u1.user_id) as monthly_active_users
from user_actions as u1
join user_actions as u2
on extract(month from u1.event_date)+1=extract(month from u2.event_date) and u1.user_id=u2.user_id and extract(month from u1.event_date)=6
group by extract(month from u2.event_date);

select distinct(s2.sub_id),(case when s1.nc is null then 0 else s1.nc end) as number_of_comments
from Submissions as s2
left join 
(
select parent_id,count(distinct(sub_id)) as nc
from Submissions as s1
group by parent_id
) AS s1
on s1.parent_id=s2.sub_id
where s2.parent_id is null
order by s2.sub_id;
